API Endpoints Documentation

/pokemon/create

Method: POST

Summary: Create a new pokemon

Request Body:

{
  "name": "Jigglypuff",
  "type": "Normal/Fairy",
  "owner": "Jiggly"
}

Responses:

201 Created: Successfully created the pokemon.

{
  "id": 16,
  "name": "Jigglypuff",
  "type": "Normal/Fairy",
  "owner": "Jiggly"
}

400 Bad Request: Missing required fields.

/pokemon

Method: GET

Summary: Get all available pokemon

Returns: A list of Pokémon

Responses:

200 OK: Successfully retrieved the list of Pokémon. 

[
  {
    "id": 1,
    "name": "Pikachu",
    "type": "Electric",
    "owner": "Ash"
  },
  {
    "id": 2,
    "name": "Charmander",
    "type": "Fire",
    "owner": "Brock"
  }
]

/pokemon/{id}

Method: GET

Summary: Get pokemon by its ID

Parameters:

{
  "id": "integer"
}

Responses:

200 OK: Successfully retrieved the pokemon.

{
  "id": 2,
  "name": "Charmander",
  "type": "Fire",
  "owner": "Brock"
}

/pokemon/delete/{id}

Method: DELETE

Summary: Delete a pokemon by ID

Parameters:

{
  "id": "integer"
}

Responses:

200 OK: Successfully deleted the pokemon.

{
  "message": "Pokemon deleted"
}

404 Not Found: Pokemon not found.

/pokemon/update/{id}

Method: PUT

Summary: Update a pokemon by ID

Parameters:

{
  "id": "integer"
}

Request Body:

{
  "name": "Raichu",
  "type": "Electric",
  "owner": "Ash"
}

Responses:

200 OK: Successfully updated the pokemon.

{
  "id": 1,
  "name": "Raichu",
  "type": "Electric",
  "owner": "Ash"
}

404 Not Found: Pokemon not found.
